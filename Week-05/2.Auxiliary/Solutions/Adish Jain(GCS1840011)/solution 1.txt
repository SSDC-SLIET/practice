#include<bits/stdc++.h>

using namespace std;

int main()
 {

	int t;

	cin>>t;

	while(t--){

	    int n,i,j;

	    cin>>n;

	    int arr[n][n];

	    for(i=0;i<n;i++)

	        for(j=0;j<n;j++)

	            cin>>arr[i][j];

	    for(i=0;i<n;i++)

	        for(j=0;j<n;j++)

	            cout<<arr[n-j-1][i]<<" ";

	    cout<<endl;

	
    }

	return 0;

}
